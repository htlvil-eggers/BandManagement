CREATE OR REPLACE TRIGGER BANDS_BEFOREINSERT 
BEFORE INSERT OR UPDATE ON BANDS 
FOR EACH ROW 
DECLARE
  maxId integer;  
BEGIN
  if :old.id <> null then
    :new.id := :old.id; 
    
  else 
    select nvl (max(id), -1) into maxId from bands;      
    :new.id := maxId + 1;   
  end if;
END;
